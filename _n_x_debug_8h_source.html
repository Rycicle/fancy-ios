00062 /*
00063  Use this when you have a variable that is only used on certain configurations (tipically DEBUG).
00064  It has zero runtime cost.
00065  
00066  Example:
00067  
00068  ----------------------------
00069  int length = [string length];
00070  ASSERT(length > 0);
00071  // If we never use length again, the compiler will complain in release builds.
00072  UNUSED(variable);
00073  ----------------------------
00074 
00075  Another use is to shut up the compiler when we don't use a parameter.
00076  
00077  */
00078 #define UNUSED(variable) ((void)variable)

00081 /*
00082  ASSERT is a nicer version of NSAssertX, with the added benefit that it does nothing on release builds.
00083  Oh, one more thing. If you're running on the debugger, you can continue. Yep, it's like a good old conditional breakpoint.
00084  */
00085 #ifndef DEBUG
00086         #define ASSERT(condition) do { } while (0);
00087 #else
00088         #define ASSERT(condition) do { if(!(condition)) { _NXAssert([NSString stringWithCString:#condition encoding:NSUTF8StringEncoding], __FILE__, __LINE__, __PRETTY_FUNCTION__); } } while (0);
00089 #endif

00091 /*
00092  TRACE is a nicer version of NSLog, with the added benefit that it does nothing on release builds.
00093  
00094  Usage: 
00095  
00096  TRACE();
00097  TRACE(@"Hello world");
00098  TRACE(@"The value is: %d", value);
00099  TRACE("The value is: %d", value);
00100  
00101  */
00102 #ifndef DEBUG
00103 #define TRACE(...) do { } while (0);
00104 #else
00105 #define TRACE(format, ...) do { _NXTrace((@"%s:%d:1 [%s] " format), __FILE__, __LINE__, __PRETTY_FUNCTION__, ##__VA_ARGS__); } while (0);
00106 #endif
